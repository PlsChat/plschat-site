async function sendLicense() { if (!serialPort || !currentLicenseKey) { alert("Device not connected or no licence found."); return; } const box = document.getElementById("statusBox"); box.innerHTML += "\n→ Sending licence to device..."; try { const writer = serialPort.writable.getWriter(); const command = `LICENSE:${currentLicenseKey}\n`; await writer.write(new TextEncoder().encode(command)); writer.releaseLock(); box.innerHTML += "\n✅ Licence command sent!"; box.innerHTML += "\nWaiting for device response..."; // Read response with timeout, accumulating chunks const reader = serialPort.readable.getReader(); let response = ''; const timeout = new Promise(resolve => setTimeout(resolve, 5000)); try { while (true) { const readPromise = reader.read(); const result = await Promise.race([readPromise, timeout.then(() => ({ done: true, value: null }))]); if (result.done || result.value === null) break; response += new TextDecoder().decode(result.value); box.innerHTML += `\nDevice: ${response.trim()}`; // Check for known OK responses if (response.includes("LICENCE_OK") || response.includes("LICENSE_OK") || response.includes("OK")) { box.innerHTML += "\n🎉 Device accepted the licence!"; break; } if (response.includes("FAIL") || response.includes("ERROR") || response.includes("INVALID")) { box.innerHTML += "\n❌ Device rejected the licence."; break; } } } finally { reader.releaseLock(); } } catch (e) { box.innerHTML += `\n❌ Failed to send licence: ${e.message}`; } }
